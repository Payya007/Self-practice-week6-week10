let quotes = [] // สร้าง [] ปล่าว

const quoteList = document.getElementById("quote-list") //  เอาตัว quoteList มา


function renderQuotes() {
       quoteList.innerHTML = ""  // เคลียร์ obj
       quotes.forEach(q => {
                 const p = document.createElement("p") // สร้าง p
                p.textContent = `"${q.content}" - ${q.author}`// รับค่า content , author มา
                quoteList.appendChild(p) // มา append มาแสดงผล
       })
}

function addQuote(content, author){
         const quote = {content, author} // อัพเดทค่า content , author
            quotes.push(quote) // push เข้าไป
            return quote
}
addQuote('Stay hungry, stay foolish.', 'Steve Jobs')
