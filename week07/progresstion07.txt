const pDiv = document.documentElement.children[1].children[0];  // 1 คือ body   และ  0 คือ ตัวแรก เจอ div เลย 
console.log(pDiv);  // เจอ div ดิ
const pText = document.documentElement.children[1].children[0].children[0]; // 0 คือ ตัวแรก เจอ p เลย
console.log(pText);  // เจอ p 

 const pElement = document.createElement("p"); 
pElement.setAttribute("format", "italic"); // set format  italic
document.getElementById("demo").appendChild(pElement); // ต่อ id demo

const base = document.createElement("i"); // สร้าง i 

const inHtml = base.cloneNode(); //  สร้าง <i></i> ใหม่จาก base
inHtml.innerHTML = "Sample Italic text "; // ใส่ข้อความ
pElement.appendChild(inHtml); // ต่อไป 

const inText = base.cloneNode();  //  สร้าง <i></i> ใหม่จาก base
inText.innerText = "Sample Italic text "; // ใส่ข้อความ
pElement.appendChild(inText); // ต่อไป

const textCon = base.cloneNode(); //  สร้าง <i></i> ใหม่จาก base
textCon.textContent = "Sample Italic text "; // ใส่ข้อความ
pElement.appendChild(textCon); // ต่อไป




Child คือ อยู่ข้างใน root  เช่น head และ body เป็นลูกของ html
Parent  คือ ตัวที่ครอบลูก เช่น html เป็น parent ของ body
Sibling  คือ nodes ที่มี parent เดียวกันเช่น head กับ body เป็น sibling กัน เพราะทั้งคู่เป็นลูกของ html
